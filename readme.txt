for c program:  1. compile: gcc compression_tool.c -o compression_tool

		   chmod +x calculate_compression_ratio.sh
		2. ./compress.sh input.txt compressed.txt
		   ./file_size.sh input.txt
		   ./file_size.sh compressed.txt
		   ./calculate_compression_ratio.sh input.txt compressed.txt
		   ./decompress.sh input.txt decompressed.txt
